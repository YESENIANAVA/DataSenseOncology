    <script>
        lucide.createIcons();

        // Cargar pacientes desde la Base de Datos (Jinja2)
        const pacientesDB = [
            {% for p in pacientes %}
            { id: {{ p.id }}, nombre: "{{ p.nombre_completo }}" },
            {% endfor %}
        ];

        // Función para filtrar y mostrar sugerencias
        function filtrarPacientes() {
            const input = document.getElementById('buscadorPaciente').value.toLowerCase();
            const lista = document.getElementById('listaResultados');
            lista.innerHTML = ''; 

            if (input.length < 1) {
                lista.classList.add('hidden');
                return;
            }

            const filtrados = pacientesDB.filter(p => 
                p.nombre.toLowerCase().includes(input) || p.id.toString().includes(input)
            );

            if (filtrados.length > 0) {
                lista.classList.remove('hidden');
                filtrados.forEach(p => {
                    const li = document.createElement('li');
                    li.className = "p-3 hover:bg-pink-50 cursor-pointer border-b border-gray-100 text-sm text-gray-700 flex justify-between";
                    li.innerHTML = `<span><i data-lucide="user" class="inline w-4 h-4 mr-2 text-gray-400"></i>${p.nombre}</span> <span class="text-[10px] font-bold text-pink-600">ID: ${p.id}</span>`;
                    li.onclick = () => seleccionarPaciente(p.id, p.nombre);
                    lista.appendChild(li);
                });
                lucide.createIcons();
            } else {
                lista.classList.remove('hidden');
                lista.innerHTML = `<li class="p-3 text-sm text-gray-400 italic text-center">No se encontraron pacientes.</li>`;
            }
        }

        // Función al seleccionar un paciente
        function seleccionarPaciente(id, nombre) {
            document.getElementById('buscadorPaciente').value = '';
            document.getElementById('listaResultados').classList.add('hidden');
            
            // Llenar tarjeta
            document.getElementById('pacienteSeleccionado').classList.remove('hidden');
            document.getElementById('nombrePacienteLabel').innerText = nombre;
            document.getElementById('pacienteIdSecreto').value = id;
            
            // Desbloquear resto del formulario
            document.getElementById('camposTumor').classList.remove('opacity-50', 'pointer-events-none');
        }

        // Función para cambiar de paciente
        function limpiarPaciente() {
            document.getElementById('pacienteSeleccionado').classList.add('hidden');
            document.getElementById('pacienteIdSecreto').value = '';
            
            // Bloquear formulario
            document.getElementById('camposTumor').classList.add('opacity-50', 'pointer-events-none');
        }
    </script>
