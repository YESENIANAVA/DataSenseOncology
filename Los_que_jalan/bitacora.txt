    <script>
        lucide.createIcons();

        function aplicarFiltros() {
            const textFilter = document.getElementById("textSearch").value.toLowerCase();
            const dateStart = document.getElementById("dateStart").value;
            const dateEnd = document.getElementById("dateEnd").value;
            const rows = document.getElementsByClassName("log-row");
            let countVisible = 0;

            Array.from(rows).forEach(row => {
                const text = row.textContent.toLowerCase();
                const rowDateFull = row.querySelector(".log-date").innerText; // Ejemplo: "2026-02-09 14:30:00"
                const rowDate = rowDateFull.split(" ")[0]; // Solo "2026-02-09"

                // Lógica de validación
                const matchesText = text.includes(textFilter);
                const matchesStart = !dateStart || rowDate >= dateStart;
                const matchesEnd = !dateEnd || rowDate <= dateEnd;

                if (matchesText && matchesStart && matchesEnd) {
                    row.style.display = "";
                    countVisible++;
                } else {
                    row.style.display = "none";
                }
            });

            document.getElementById("noResults").classList.toggle("hidden", countVisible > 0);
        }

        function limpiarFiltros() {
            document.getElementById("textSearch").value = "";
            document.getElementById("dateStart").value = "";
            document.getElementById("dateEnd").value = "";
            aplicarFiltros();
        }
    </script>
