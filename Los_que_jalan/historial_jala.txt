    <script>
        lucide.createIcons();
        
        // ==========================================
        // 1. FECHA DE IMPRESIÓN (Corregido)
        // Verificamos si existe el elemento antes de ponerle la fecha para no romper JS
        // ==========================================
        const elementoFecha = document.getElementById('fecha-impresion');
        if (elementoFecha) {
            elementoFecha.innerText = "Impreso el: " + new Date().toLocaleDateString('es-ES');
        }

        // ==========================================
        // 2. CÁLCULO DE PREDICCIÓN TEÓRICA AUTOMÁTICO
        // ==========================================
        {% if tumor %}
        document.addEventListener("DOMContentLoaded", function() {
            const largo = parseFloat("{{ tumor.largo or tumor.dimension_largo or 0 }}");
            const ancho = parseFloat("{{ tumor.ancho or tumor.dimension_ancho or 0 }}");
            const prof = parseFloat("{{ tumor.profundidad or tumor.dimension_profundidad or 0 }}");
            
            if(largo > 0 && ancho > 0 && prof > 0) {
                const volumenInicial = (4/3) * Math.PI * (largo/2) * (ancho/2) * (prof/2);
                document.getElementById('vol-teorico-celda').innerText = volumenInicial.toFixed(2) + " mm³";
                
                const proy30 = volumenInicial * Math.exp(0.015 * 30);
                document.getElementById('proy-teorica-celda').innerText = proy30.toFixed(2) + " mm³";
            } else {
                document.getElementById('vol-teorico-celda').innerText = "Dimensión faltante";
                document.getElementById('proy-teorica-celda').innerText = "N/A";
            }
        });
        {% endif %}

        // ==========================================
        // 3. GRÁFICA REAL DE CHART.JS
        // ==========================================
        {% if predicciones and predicciones|length > 0 %}
        document.addEventListener("DOMContentLoaded", function() {
            const ctx = document.getElementById('graficaHistorial').getContext('2d');
            
            const fechas = [ {% for p in predicciones %} "{{ p.fecha_registro }}", {% endfor %} ];
            const volumenes = [ {% for p in predicciones %} {{ p.volumen_actual }}, {% endfor %} ];

            new Chart(ctx, {
                type: 'line',
                data: {
                    labels: fechas,
                    datasets: [{
                        label: 'Evolución del Volumen Tumoral (mm³)',
                        data: volumenes,
                        borderColor: '#1f2937', // Gris Institucional
                        backgroundColor: '#1f2937',
                        borderWidth: 2,
                        tension: 0.3,
                        pointRadius: 4
                    }]
                },
                options: {
                    responsive: true,
                    plugins: { legend: { display: false } },
                    scales: {
                        x: { ticks: { font: {size: 10} } },
                        y: { ticks: { font: {size: 10} } }
                    }
                }
            });
        });
        {% endif %}

        // ==========================================
        // 4. LÓGICA DEL BUSCADOR (Restaurado y Protegido)
        // ==========================================
        const pacientesDB = [
            {% for p in pacientes %}
            { id: {{ p.id }}, nombre: "{{ p.nombre_completo }}" },
            {% endfor %}
        ];

        function filtrarPacientes() {
            const input = document.getElementById('buscadorPaciente').value.toLowerCase();
            const lista = document.getElementById('listaResultados');
            lista.innerHTML = ''; 

            if (input.length < 1) {
                lista.classList.add('hidden');
                return;
            }

            const filtrados = pacientesDB.filter(p => p.nombre.toLowerCase().includes(input) || p.id.toString().includes(input));

            if (filtrados.length > 0) {
                lista.classList.remove('hidden');
                filtrados.forEach(p => {
                    const li = document.createElement('li');
                    li.className = "p-3 hover:bg-gray-100 cursor-pointer border-b border-gray-200 text-sm flex justify-between items-center";
                    li.innerHTML = `<span><i data-lucide="user" class="inline w-4 h-4 mr-2"></i>${p.nombre}</span> <span class="text-[10px] bg-gray-200 text-gray-800 px-2 py-0.5 rounded">ID: ${p.id}</span>`;
                    li.onclick = () => { window.location.href = `/historial_medico?user={{ doctor.username }}&paciente_id=${p.id}`; };
                    lista.appendChild(li);
                });
                lucide.createIcons();
            } else {
                // Restauramos el mensaje de "No encontrado"
                lista.classList.remove('hidden');
                lista.innerHTML = `<li class="p-3 text-sm text-gray-400 italic text-center">No se encontraron pacientes.</li>`;
            }
        }

        // ==========================================
        // 5. EXPORTACIÓN A PDF FORMATO A4 VERTICAL
        // ==========================================
        function generarPDF(nombrePaciente) {
            const elemento = document.getElementById('area-imprimible');
            const botonPDF = document.getElementById('btn-pdf');
            
            if(botonPDF) botonPDF.style.display = 'none';

            const opciones = {
                margin:       10, 
                filename:     `Expediente_Clinico_${nombrePaciente.replace(/\s+/g, '_')}.pdf`,
                image:        { type: 'jpeg', quality: 0.98 },
                html2canvas:  { scale: 2, useCORS: true }, 
                jsPDF:        { unit: 'mm', format: 'a4', orientation: 'portrait' }
            };

            html2pdf().set(opciones).from(elemento).save().then(() => {
                if(botonPDF) botonPDF.style.display = 'block';
            });
        }
