        function filtrarPacientes() {
            const input = document.getElementById('buscadorPaciente').value.toLowerCase();
            const lista = document.getElementById('listaResultados');
            lista.innerHTML = ''; 

            if (input.length < 1) {
                lista.classList.add('hidden');
                return;
            }

            const filtrados = pacientesDB.filter(p => 
                p.nombre.toLowerCase().includes(input) || p.id.toString().includes(input)
            );

            if (filtrados.length > 0) {
                lista.classList.remove('hidden');
                filtrados.forEach(p => {
                    const li = document.createElement('li');
                    li.className = "p-3 hover:bg-pink-50 cursor-pointer border-b border-gray-100 text-sm text-gray-700 flex justify-between";
                    li.innerHTML = `<span><i data-lucide="user" class="inline w-4 h-4 mr-2 text-gray-400"></i>${p.nombre}</span> <span class="text-[10px] font-bold text-pink-600">ID: ${p.id}</span>`;
                    li.onclick = () => seleccionarPaciente(p.id, p.nombre);
                    lista.appendChild(li);
                });
                lucide.createIcons();
            } else {
                lista.classList.remove('hidden');
                lista.innerHTML = `<li class="p-3 text-sm text-gray-400 italic text-center">No se encontraron pacientes.</li>`;
            }
        }

        function seleccionarPaciente(id, nombre) {
            document.getElementById('buscadorPaciente').value = '';

            // Mostrar tarjeta

            document.getElementById('nombrePacienteLabel').innerText = nombre;
            document.getElementById('pacienteIdSecreto').value = id;
            
            // Desbloquear tabla
            document.getElementById('seccionDatos').classList.remove('opacity-50', 'pointer-events-none');
        }

        async function procesarPrediccion() {
            const pacienteId = document.getElementById('pacienteIdSecreto').value;
            const filas = document.querySelectorAll('.fila-medicion');
            
            let datos = [];
            let valid = true;

            filas.forEach(fila => {
                const fecha = fila.querySelector('input[type="date"]').value;
                const volumen = fila.querySelector('input[type="number"]').value;
                const enTratamiento = fila.querySelector('.chk-tratamiento').checked;
                if(!fecha || !volumen) valid = false;
                datos.push({fecha: fecha, volumen: parseFloat(volumen), tratamiento: enTratamiento});
            });

            if(!valid) {
                alert("Por favor, complete todas las fechas y volúmenes antes de calcular.");
                return;
            }
            if(datos.length < 2) {
                alert("Se necesitan al menos 2 mediciones para calcular el crecimiento.");
                return;
            }

            // Cambiar botón a estado de carga
            const btn = document.getElementById('btnCalcular');
            const originalText = btn.innerHTML;
            btn.innerHTML = `<i data-lucide="loader-2" class="w-5 h-5 animate-spin"></i> Ejecutando Algoritmo...`;
            btn.disabled = true;

            try {
                // Enviamos los datos a Python
                const response = await fetch('/calcular_prediccion_real', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        paciente_id: parseInt(pacienteId),
                        historial: datos
                    })
                });

                const resultado = await response.json();

                if(resultado.error) {
                    alert("Error matemático: " + resultado.error);
                } else {
                    mostrarResultados(resultado);
                }
            } catch (error) {
                console.error(error);
                alert("Fallo de conexión con el motor IA.");
            } finally {
                btn.innerHTML = originalText;
                btn.disabled = false;
                lucide.createIcons();
            }
        }

        let chartInstance = null; // Para no encimar gráficas si le dan clic varias veces

        function dibujarGrafica() {
            // Mostramos el cuadro blanco
            document.getElementById('contenedorGrafica').classList.remove('hidden');

            // 1. Extraemos las fechas y volúmenes de la tabla que llenó el médico
            const filas = document.querySelectorAll('.fila-medicion');
            let labels = [];
            let dataReales = [];

            filas.forEach(fila => {
                labels.push(fila.querySelector('input[type="date"]').value);
                dataReales.push(parseFloat(fila.querySelector('input[type="number"]').value));
            });

            // 2. Preparamos las etiquetas a futuro (+30, +60, +90 días)
            labels.push('+30 Días', '+60 Días', '+90 Días');
            
            // 3. Empalmamos la línea de predicción con el último punto real
            let dataPrediccion = Array(dataReales.length - 1).fill(null);
            dataPrediccion.push(dataReales[dataReales.length - 1]); // Conecta la realidad con la predicción
            dataPrediccion.push(ultimosResultados.prediccion_30_dias);
            dataPrediccion.push(ultimosResultados.prediccion_60_dias);
            dataPrediccion.push(ultimosResultados.prediccion_90_dias);

            // 4. Dibujamos la gráfica
            const ctx = document.getElementById('miGrafica').getContext('2d');
            if(chartInstance) chartInstance.destroy(); // Borra gráfica vieja si existe

            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [
                        {
                            label: 'Volumen Clínico Registrado (mm³)',
                            data: dataReales,
                            borderColor: '#be185d', // Rosa oscuro
                            backgroundColor: '#be185d',
                            borderWidth: 3,
                            tension: 0.3,
                            pointRadius: 6
                        },
                        {
                            label: 'Proyección Predictiva IA (mm³)',
                            data: dataPrediccion,
                            borderColor: '#ef4444', // Rojo punteado
                            borderDash: [5, 5],
                            borderWidth: 3,
                            tension: 0.3,
                            pointRadius: 6,
                            pointBackgroundColor: '#ef4444'
                        }
                    ]
                },
                options: { 
                    responsive: true, 
                    plugins: { legend: { position: 'bottom' } } 
                }
            });

            // Bajamos la pantalla suavemente hacia la gráfica
            document.getElementById('contenedorGrafica').scrollIntoView({ behavior: 'smooth', block: 'end' });
        }

        async function guardarEnHistorial() {
            // Buscamos el botón que se presionó
            const btn = document.querySelector('button[onclick="guardarEnHistorial()"]');
            const originalText = btn.innerHTML;
            
            // Efecto visual de carga
            btn.innerHTML = `<i data-lucide="loader-2" class="w-4 h-4 animate-spin"></i> Guardando...`;
            btn.disabled = true;

            const pacienteId = document.getElementById('pacienteIdSecreto').value;

            try {
                // Enviamos los datos a Python de forma invisible
                const response = await fetch('/guardar_prediccion_historial', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        paciente_id: parseInt(pacienteId),
                        dias_analizados: ultimosResultados.dias_analizados,
                        tasa_crecimiento: ultimosResultados.tasa_diaria_porcentaje,
                        volumen_actual: ultimosResultados.volumen_actual,
                        tendencia: ultimosResultados.tendencia,
                        medico_operando: "{{ doctor.username }}" // Inyectamos quién está logueado
                    })
                });

                const resultado = await response.json();

                if(resultado.status === "success") {
                    // Si todo sale bien, pintamos el botón de gris oscuro y le ponemos palomita
                    btn.classList.replace('bg-green-600', 'bg-gray-800');
                    btn.classList.replace('hover:bg-green-500', 'hover:bg-gray-700');
                    btn.innerHTML = `<i data-lucide="check-circle" class="w-4 h-4"></i> ¡Guardado Exitoso!`;
                    lucide.createIcons();
                    
                    // Esperamos 1.5 segundos para que el doctor vea el mensaje de éxito, y lo redirigimos a su historial
                    setTimeout(() => {
                        window.location.href = `/historial_medico?user={{ doctor.username }}&paciente_id=${pacienteId}`;
                    }, 1500);
                }

            } catch (error) {
                console.error("Error al guardar:", error);
                alert("Hubo un error de conexión al intentar guardar en la base de datos.");
                btn.innerHTML = originalText;
                btn.disabled = false;
                lucide.createIcons();
            }
        }
