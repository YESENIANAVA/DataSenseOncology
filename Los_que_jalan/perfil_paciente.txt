    <script>
        lucide.createIcons();

        async function ejecutarIA(pacienteId) {
            document.getElementById('ia-initial').classList.add('hidden');
            document.getElementById('ia-loader').classList.remove('hidden');
            document.getElementById('cpu-icon').classList.add('animate-spin');

            try {
                // Esta ruta llamará a la lógica que NO usa fechas
                const response = await fetch(`/ejecutar_ia/${pacienteId}`);
                const data = await response.json();

                if (data.error) {
                    alert("Error: " + data.error);
                    location.reload();
                    return;
                }

                setTimeout(() => {
                    document.getElementById('ia-loader').classList.add('hidden');
                    document.getElementById('ia-results').classList.remove('hidden');
                    document.getElementById('cpu-icon').classList.remove('animate-spin');

                    // Mostramos los resultados basados en la Lógica A (Fórmula Exponencial)
                    document.getElementById('vol-actual').innerText = `${data.volumen_actual} mm³`;
                    document.getElementById('vol-predicho').innerText = `${data.volumen_predicho} mm³`;
                    document.getElementById('pct-incremento').innerText = `+${data.incremento_porcentaje}%`;
                }, 1000);

            } catch (error) {
                console.error("Error:", error);
                alert("Hubo un problema al conectar con el servidor.");
                location.reload();
            }
        }
    </script>
